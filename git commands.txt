cd [directory] - change directory to the directory in []
git init - initialize directory for cloud sharing
git remote add [name] [remote-url] - add remote location to the local branch
git add [file] - add file to local branch. '.' to add all
git commit -m [message] - commit local changes and prepare for upload. message <= 50 chars
git push [remote] [branch] - upload to the [remote] location on the [branch] branch 
git fetch - searches for and downloads changes to cache 
git pull [remote] [branch] - overwrites files
git status - lists all changes since last commit
git branch [name] - creates new branch
git branch {--remote} - --remote lists remote branches otherwise lists local branches 
git checkout [name] - download new branch/switch to another branch
